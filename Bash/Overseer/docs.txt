  -f | -files – опция для работы с файлами
  -f -c | -cat | -read <file_path> – читает исходный код файла
  -f -h | -head <file_path> – читает первые 20 линий файла
  -f -u | -unpack | -unzip <file_path> – распаковывает ZIP, APK, DEX файлы
  -f -uid | -search_uid <uid> – получает имя юзера по его UID
  -f -s | -stats <file_patg> – подробная информация о файле: Inode, Владелец (имя и UID), Разрешения, SHA256 Хеш, энтропия и многое другое 
   
  -w | -web – опция для работы с web ресурсами
  -w -d | -domain <domain> – получает информацию о домене (айпи адреса, AAAA, MX и т.д)
  -w -i | -ip <IPv4> – получает получает информацию о данном айпи адресе. Если ты ничего не вводишь, то отображается твоя информация
  -w -wget | -savebody <URL> <FileName> – сохраняет результат запроса с таким названием файла

  -self – опция, которая получает информацию о твоём устройстве (RAM, Кол-во процессоров, Имя ОС и платформы, архитектура и т.д)
  
  -t | -terminal | -env – опция для получения информации о текущем экзекуторе
  -t -s | -source <command> – получает исходный код команды (если это возможно)
  -t -print <text> <ASNI_Code> – печатает текст с заданным цветным кодом (например 31 это красный). Если ты не введёшь ASNI код, то текст будет белым
  -t -c | -clear – эквивалентно обычному clear
  -t -close <code> – эквивалентно обычному exit
  -t -cmds | -commands – получает все команды (большинство)

  -ex | -execute – опция, которая позволяет запускать разные языки программирования и выводить stdout (или stderr если есть) из файла или из текста
  -ex -py | -python <code_str_or_file> <args> – запускает Python скрипт
  -ex -c <code_str_or_file> – запускает C скрипт
  -ex -cpp <code_str_or_file> – запускает C++ скрипт
  -ex -sh | -shell | -bash <code_str_or_file> – запускает Shell (Bash) скрипт
  -ex -java <code_str_or_file> – запускает Java скрипт
  -ex -node | -njs <code_str_or_file> – запускает Node.js скрипт 
  -ex -lua <code_str_or_file> – запускает Lua скрипт

  --version – просто получает версию данной команды (например 1.0)
  --help – показывает всё то, что сейчас тут написано
